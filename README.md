import java.util.Scanner; public class CalculadoraCientifica { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); int opcion; do { System.out.println("\n=== CALCULADORA CIENTÍFICA 🧮 ==="); System.out.println("1.
